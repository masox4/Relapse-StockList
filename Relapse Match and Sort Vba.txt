Option Explicit

'========================
' Helpers
'========================
Private Function NormalizeText(ByVal s As String) As String
    s = Replace(s, Chr(160), " ")
    s = Trim$(s)
    Do While InStr(s, "  ") > 0
        s = Replace(s, "  ", " ")
    Loop
    NormalizeText = UCase$(s)
End Function

Private Function GetSheetLoose(ByVal wb As Workbook, ByVal sheetName As String) As Worksheet
    Dim sh As Object
    Dim target As String

    target = NormalizeText(sheetName)

    For Each sh In wb.Sheets
        If NormalizeText(sh.Name) = target Then
            Set GetSheetLoose = sh
            Exit Function
        End If
    Next sh
End Function

Private Function SkuSizeRank(ByVal sku As String) As Long
    sku = UCase$(Trim$(sku))

    If Right$(sku, 4) = "XXXX" Then
        SkuSizeRank = 7
    ElseIf Right$(sku, 3) = "XXX" Then
        SkuSizeRank = 6
    ElseIf Right$(sku, 2) = "XX" Then
        SkuSizeRank = 5
    ElseIf Right$(sku, 1) = "X" Then
        SkuSizeRank = 4
    ElseIf Right$(sku, 1) = "L" Then
        SkuSizeRank = 3
    ElseIf Right$(sku, 1) = "M" Then
        SkuSizeRank = 2
    ElseIf Right$(sku, 1) = "S" Then
        SkuSizeRank = 1
    Else
        SkuSizeRank = 0
    End If
End Function

Private Function SkuBase(ByVal sku As String) As String
    Dim u As String

    u = UCase$(Trim$(sku))

    If Right$(u, 4) = "XXXX" Then
        SkuBase = Left$(u, Len(u) - 4)
    ElseIf Right$(u, 3) = "XXX" Then
        SkuBase = Left$(u, Len(u) - 3)
    ElseIf Right$(u, 2) = "XX" Then
        SkuBase = Left$(u, Len(u) - 2)
    ElseIf Right$(u, 1) = "X" Or Right$(u, 1) = "L" Or Right$(u, 1) = "M" Or Right$(u, 1) = "S" Then
        SkuBase = Left$(u, Len(u) - 1)
    Else
        SkuBase = u
    End If
End Function

Private Sub EnsureParentAndSmallRows(ByVal ws As Worksheet, ByVal colSkuA As Long, ByVal colSkuE As Long, ByVal headerRow As Long)
    Dim lastRow As Long
    Dim r As Long
    Dim sku As String, baseSku As String
    Dim dictHasParent As Object
    Dim dictHasSmall As Object

    Set dictHasParent = CreateObject("Scripting.Dictionary")
    dictHasParent.CompareMode = vbTextCompare

    Set dictHasSmall = CreateObject("Scripting.Dictionary")
    dictHasSmall.CompareMode = vbTextCompare

    lastRow = ws.Cells(ws.Rows.Count, colSkuA).End(xlUp).Row

    For r = headerRow + 1 To lastRow
        sku = Trim$(CStr(ws.Cells(r, colSkuA).Value))
        If Len(sku) > 0 Then
            baseSku = SkuBase(sku)
            If SkuSizeRank(sku) = 0 Then
                dictHasParent(baseSku) = True
            ElseIf SkuSizeRank(sku) = 1 Then
                dictHasSmall(baseSku) = True
            End If
        End If
    Next r

    For r = lastRow To headerRow + 1 Step -1
        sku = Trim$(CStr(ws.Cells(r, colSkuA).Value))
        If Len(sku) > 0 Then
            baseSku = SkuBase(sku)

            If SkuSizeRank(sku) > 0 Then
                If Not dictHasParent.Exists(baseSku) Then
                    ws.Rows(r).Insert Shift:=xlShiftDown
                    ws.Cells(r, colSkuA).Value = baseSku
                    ws.Cells(r, colSkuE).Value = baseSku
                    dictHasParent(baseSku) = True
                End If

                If Not dictHasSmall.Exists(baseSku) Then
                    ws.Rows(r + 1).Insert Shift:=xlShiftDown
                    ws.Cells(r + 1, colSkuA).Value = baseSku & "S"
                    ws.Cells(r + 1, colSkuE).Value = baseSku & "S"
                    dictHasSmall(baseSku) = True
                End If
            End If
        End If
    Next r
End Sub

Private Sub SortBySkuFamily(ByVal ws As Worksheet, ByVal headerRow As Long, ByVal colSkuA As Long, ByVal lastDataCol As Long)
    Dim lastRow As Long
    Dim r As Long
    Dim colHelperBase As Long, colHelperRank As Long
    Dim sku As String

    lastRow = ws.Cells(ws.Rows.Count, colSkuA).End(xlUp).Row
    If lastRow <= headerRow + 1 Then Exit Sub

    colHelperBase = lastDataCol + 1
    colHelperRank = lastDataCol + 2

    ws.Cells(headerRow, colHelperBase).Value = "_BASE"
    ws.Cells(headerRow, colHelperRank).Value = "_RANK"

    For r = headerRow + 1 To lastRow
        sku = Trim$(CStr(ws.Cells(r, colSkuA).Value))
        ws.Cells(r, colHelperBase).Value = SkuBase(sku)
        ws.Cells(r, colHelperRank).Value = SkuSizeRank(sku)
    Next r

    With ws.Sort
        .SortFields.Clear
        .SortFields.Add Key:=ws.Range(ws.Cells(headerRow + 1, colHelperBase), ws.Cells(lastRow, colHelperBase)), _
                        SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=ws.Range(ws.Cells(headerRow + 1, colHelperRank), ws.Cells(lastRow, colHelperRank)), _
                        SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SetRange ws.Range(ws.Cells(headerRow, 1), ws.Cells(lastRow, colHelperRank))
        .Header = xlYes
        .MatchCase = False
        .Orientation = xlTopToBottom
        .Apply
    End With

    ws.Columns(colHelperBase).ClearContents
    ws.Columns(colHelperRank).ClearContents
End Sub

'========================
' Main macro (does BOTH tasks)
'========================
Public Sub Run_CopyHere_ParentRows_And_CategoryCheck()
    Const COL_SKU_A As Long = 1
    Const COL_BAND_C As Long = 3
    Const COL_SKU_E As Long = 5
    Const COL_OUT_M As Long = 13
    Const HEADER_ROW As Long = 1

    Const CAT_COL_G As Long = 7
    Const CAT_COL_J As Long = 10

    Dim wb As Workbook
    Dim wsCopy As Worksheet, wsCat As Worksheet
    Dim lastRow As Long, r As Long

    Dim dictCatTerms As Object
    Dim bandTerm As String
    Dim v As String
    Dim lastG As Long, lastJ As Long, lastCat As Long

    Set wb = ActiveWorkbook
    Set wsCopy = GetSheetLoose(wb, "Copy Here")
    Set wsCat = GetSheetLoose(wb, "CatMap")

    If wsCopy Is Nothing Or wsCat Is Nothing Then
        MsgBox "Missing one or more sheets: Copy Here / CatMap. Check sheet names.", vbExclamation
        Exit Sub
    End If

    On Error GoTo CleanFail
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.Calculation = xlCalculationManual

    EnsureParentAndSmallRows wsCopy, COL_SKU_A, COL_SKU_E, HEADER_ROW
    SortBySkuFamily wsCopy, HEADER_ROW, COL_SKU_A, COL_OUT_M

    Set dictCatTerms = CreateObject("Scripting.Dictionary")
    dictCatTerms.CompareMode = vbTextCompare

    lastG = wsCat.Cells(wsCat.Rows.Count, CAT_COL_G).End(xlUp).Row
    lastJ = wsCat.Cells(wsCat.Rows.Count, CAT_COL_J).End(xlUp).Row
    lastCat = IIf(lastG > lastJ, lastG, lastJ)

    For r = 2 To lastCat
        v = NormalizeText(CStr(wsCat.Cells(r, CAT_COL_G).Value))
        If Len(v) > 0 Then dictCatTerms(v) = True

        v = NormalizeText(CStr(wsCat.Cells(r, CAT_COL_J).Value))
        If Len(v) > 0 Then dictCatTerms(v) = True
    Next r

    lastRow = wsCopy.Cells(wsCopy.Rows.Count, COL_SKU_A).End(xlUp).Row

    For r = 2 To lastRow
        bandTerm = NormalizeText(CStr(wsCopy.Cells(r, COL_BAND_C).Value))
        If Len(bandTerm) = 0 Then
            wsCopy.Cells(r, COL_OUT_M).Value = vbNullString
        ElseIf dictCatTerms.Exists(bandTerm) Then
            wsCopy.Cells(r, COL_OUT_M).Value = "OK"
        Else
            wsCopy.Cells(r, COL_OUT_M).Value = "NO MATCH - CHECK CATEGORIES"
        End If
    Next r

CleanExit:
    Application.Calculation = xlCalculationAutomatic
    Application.EnableEvents = True
    Application.ScreenUpdating = True
    Exit Sub

CleanFail:
    Resume CleanExit
End Sub

'========================
' Button creator (run once)
'========================
Public Sub Create_CopyHere_RunButton()
    Dim wb As Workbook
    Dim ws As Worksheet
    Dim rng As Range
    Dim shp As Shape
    Dim nm As String: nm = "btnRunParentCat"

    Set wb = ActiveWorkbook
    Set ws = GetSheetLoose(wb, "Copy Here")
    If ws Is Nothing Then
        MsgBox "Can't find sheet 'Copy Here'.", vbExclamation
        Exit Sub
    End If

    Set rng = ws.Range("N1")

    On Error Resume Next
    ws.Shapes(nm).Delete
    On Error GoTo 0

    Set shp = ws.Shapes.AddShape(msoShapeRoundedRectangle, rng.Left + 2, rng.Top + 2, 10, 10)
    shp.Name = nm
    shp.OnAction = "Run_CopyHere_ParentRows_And_CategoryCheck"

    With shp
        .TextFrame2.TextRange.Text = " Run Parent Adder and Cat Match "
        .TextFrame2.Orientation = msoTextOrientationHorizontal
        .TextFrame2.WordWrap = msoFalse
        .TextFrame2.AutoSize = msoAutoSizeShapeToFitText
        .TextFrame2.MarginLeft = 6
        .TextFrame2.MarginRight = 6
        .TextFrame2.MarginTop = 3
        .TextFrame2.MarginBottom = 3
        .Left = rng.Left + 2
        .Top = ws.Rows(1).Top + 1

        .Fill.ForeColor.RGB = RGB(0, 51, 102)
        .Line.Visible = msoFalse

        With .TextFrame2.TextRange.Font
            .Fill.ForeColor.RGB = RGB(255, 255, 255)
            .Bold = msoTrue
            .Size = 9
        End With

        .Placement = xlMoveAndSize
    End With
End Sub
