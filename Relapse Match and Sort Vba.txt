Option Explicit

'========================
' Helpers
'========================
Private Function NormalizeText(ByVal s As String) As String
    s = Replace(s, Chr(160), " ") ' NBSP -> space
    s = Trim$(s)
    Do While InStr(s, "  ") > 0
        s = Replace(s, "  ", " ")
    Loop
    NormalizeText = UCase$(s)
End Function

Private Function SizeSortRank(ByVal sizeVal As String) As Long
    Select Case NormalizeText(sizeVal)
        Case "", "PARENT": SizeSortRank = 0
        Case "S": SizeSortRank = 1
        Case "M": SizeSortRank = 2
        Case "L": SizeSortRank = 3
        Case "XL", "X": SizeSortRank = 4
        Case "2XL", "XXL", "2X": SizeSortRank = 5
        Case "3XL", "XXXL", "3X": SizeSortRank = 6
        Case "4XL", "XXXXL", "4X": SizeSortRank = 7
        Case Else: SizeSortRank = 99
    End Select
End Function

Private Function SizeSuffixCode(ByVal sizeVal As String) As String
    Select Case NormalizeText(sizeVal)
        Case "S": SizeSuffixCode = "S"
        Case "M": SizeSuffixCode = "M"
        Case "L": SizeSuffixCode = "L"
        Case "XL", "X": SizeSuffixCode = "X"
        Case "2XL", "XXL", "2X": SizeSuffixCode = "XX"
        Case "3XL", "XXXL", "3X": SizeSuffixCode = "XXX"
        Case "4XL", "XXXXL", "4X": SizeSuffixCode = "XXXX"
        Case Else: SizeSuffixCode = vbNullString
    End Select
End Function

Private Function GetBaseSku(ByVal sku As String, ByVal sizeVal As String) As String
    Dim suffix As String
    sku = Trim$(sku)
    If Len(sku) = 0 Then
        GetBaseSku = vbNullString
        Exit Function
    End If

    suffix = SizeSuffixCode(sizeVal)
    If Len(suffix) > 0 And Len(sku) > Len(suffix) Then
        If UCase$(Right$(sku, Len(suffix))) = suffix Then
            GetBaseSku = Left$(sku, Len(sku) - Len(suffix))
            Exit Function
        End If
    End If

    GetBaseSku = sku
End Function

Private Function GetSheetLoose(ByVal wb As Workbook, ByVal sheetName As String) As Worksheet
    Dim sh As Object
    Dim target As String: target = NormalizeText(sheetName)

    For Each sh In wb.Sheets
        If NormalizeText(sh.Name) = target Then
            Set GetSheetLoose = sh
            Exit Function
        End If
    Next sh
End Function

Private Function ExtractBeforeSpaceApostrophe(ByVal fullName As String) As String
    ' Return normalized text before apostrophe marker.
    ' Primary split: " '" (space + apostrophe); fallback split: first apostrophe.
    Dim p As Long
    fullName = NormalizeText(fullName)
    fullName = Replace(fullName, "’", "'")

    p = InStr(1, fullName, " '", vbBinaryCompare)
    If p = 0 Then p = InStr(1, fullName, "'", vbBinaryCompare)

    If p > 0 Then
        ExtractBeforeSpaceApostrophe = NormalizeText(Left$(fullName, p - 1))
    Else
        ExtractBeforeSpaceApostrophe = fullName
    End If
End Function

'========================
' Main macro (does BOTH tasks)
'========================
Public Sub Run_CopyHere_ParentRows_And_CategoryCheck()
    '---- Column setup (adjust if needed)
    Const COL_SKU_A As Long = 1   'Copy Here: A
    Const COL_SKU_E As Long = 5   'Copy Here: E
    Const COL_NAME_C As Long = 3  'Copy Here: C
    Const COL_SIZE  As Long = 8   'Copy Here: H
    Const COL_OUT_M As Long = 13  'Copy Here: M
    Const HEADER_ROW As Long = 1
    Const HELPER_BLOCK_COL As Long = 702  'ZZ
    Const HELPER_SIZE_COL As Long = 703   'AAA
    Const HELPER_SEQ_COL As Long = 704    'AAB


    Const CATMAP_COL_PRIMARY_NAME_G As Long = 7   'CatMap: G
    Const CATMAP_COL_ALT_NAMES_J As Long = 10     'CatMap: J (semicolon-separated aliases)

    Dim wb As Workbook
    Dim wsCopy As Worksheet, wsCatMap As Worksheet
    Dim lastRow As Long, r As Long

    Dim dictCatMapPrimaryTerms As Object   'primary term from CatMap G -> True
    Dim dictCatMapAltTerms As Object       'fallback alias from CatMap J -> True
    Dim dictBlockOrder As Object         'base sku -> first-seen block order

    Dim sku As String, baseSku As String, sizeVal As String
    Dim nm As String, term As String
    Dim v As String
    Dim altNames As String, altTerm As String
    Dim parts As Variant, part As Variant
    Dim blockOrder As Long, base As String

    Set wb = ActiveWorkbook

    Set wsCopy = GetSheetLoose(wb, "Copy Here")
    Set wsCatMap = GetSheetLoose(wb, "CatMap")

    If wsCopy Is Nothing Or wsCatMap Is Nothing Then
        MsgBox "Missing one or more sheets: Copy Here / CatMap. Check sheet names.", vbExclamation
        Exit Sub
    End If

    On Error GoTo CleanFail
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.Calculation = xlCalculationManual

    '========================
    ' 1) Insert parent rows (above Small rows)
    '========================
    lastRow = wsCopy.Cells(wsCopy.Rows.Count, COL_SKU_A).End(xlUp).Row

    For r = lastRow To HEADER_ROW + 1 Step -1
        sku = Trim$(CStr(wsCopy.Cells(r, COL_SKU_A).Value))
        If Len(sku) > 1 And UCase$(Right$(sku, 1)) = "S" Then
            sizeVal = UCase$(Trim$(CStr(wsCopy.Cells(r, COL_SIZE).Value)))
            If sizeVal = "S" Then
                baseSku = Left$(sku, Len(sku) - 1)

                'Skip if the row above is already the parent row
                If Trim$(CStr(wsCopy.Cells(r - 1, COL_SKU_A).Value)) <> baseSku _
                   Or Trim$(CStr(wsCopy.Cells(r - 1, COL_SKU_E).Value)) <> baseSku Then

                    wsCopy.Rows(r).Insert Shift:=xlShiftDown
                    wsCopy.Cells(r, COL_SKU_A).Value = baseSku
                    wsCopy.Cells(r, COL_SKU_E).Value = baseSku
                End If
            End If
        End If
    Next r

    '========================
    ' 1b) Reorder rows into contiguous parent+size blocks
    '     - Keep each base SKU together
    '     - Order inside each block: Parent, S, M, L, XL, 2XL, 3XL, 4XL
    '========================
    Set dictBlockOrder = CreateObject("Scripting.Dictionary")
    dictBlockOrder.CompareMode = vbTextCompare

    lastRow = wsCopy.Cells(wsCopy.Rows.Count, COL_SKU_A).End(xlUp).Row
    blockOrder = 0

    For r = HEADER_ROW + 1 To lastRow
        sku = CStr(wsCopy.Cells(r, COL_SKU_A).Value)
        sizeVal = CStr(wsCopy.Cells(r, COL_SIZE).Value)
        base = GetBaseSku(sku, sizeVal)

        If Len(base) = 0 Then
            wsCopy.Cells(r, HELPER_BLOCK_COL).Value = 999999
            wsCopy.Cells(r, HELPER_SIZE_COL).Value = 999
        Else
            If Not dictBlockOrder.Exists(base) Then
                blockOrder = blockOrder + 1
                dictBlockOrder(base) = blockOrder
            End If
            wsCopy.Cells(r, HELPER_BLOCK_COL).Value = dictBlockOrder(base)
            wsCopy.Cells(r, HELPER_SIZE_COL).Value = SizeSortRank(sizeVal)
        End If
        wsCopy.Cells(r, HELPER_SEQ_COL).Value = r
    Next r

    With wsCopy.Sort
        .SortFields.Clear
        .SortFields.Add Key:=wsCopy.Range(wsCopy.Cells(HEADER_ROW + 1, HELPER_BLOCK_COL), wsCopy.Cells(lastRow, HELPER_BLOCK_COL)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=wsCopy.Range(wsCopy.Cells(HEADER_ROW + 1, HELPER_SIZE_COL), wsCopy.Cells(lastRow, HELPER_SIZE_COL)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=wsCopy.Range(wsCopy.Cells(HEADER_ROW + 1, HELPER_SEQ_COL), wsCopy.Cells(lastRow, HELPER_SEQ_COL)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal

        .SetRange wsCopy.Range(wsCopy.Cells(HEADER_ROW, COL_SKU_A), wsCopy.Cells(lastRow, HELPER_SEQ_COL))
        .Header = xlYes
        .MatchCase = False
        .Orientation = xlTopToBottom
        .Apply
    End With

    wsCopy.Range(wsCopy.Cells(HEADER_ROW + 1, HELPER_BLOCK_COL), wsCopy.Cells(lastRow, HELPER_SEQ_COL)).ClearContents

    '========================
    ' 2) Build lookups from CatMap
    '    - Primary names from column G
    '    - Fallback aliases from column J (semicolon-separated)
    '========================
    Set dictCatMapPrimaryTerms = CreateObject("Scripting.Dictionary")
    dictCatMapPrimaryTerms.CompareMode = vbTextCompare

    Set dictCatMapAltTerms = CreateObject("Scripting.Dictionary")
    dictCatMapAltTerms.CompareMode = vbTextCompare

    lastRow = wsCatMap.Cells(wsCatMap.Rows.Count, CATMAP_COL_PRIMARY_NAME_G).End(xlUp).Row
    For r = 2 To lastRow
        nm = CStr(wsCatMap.Cells(r, CATMAP_COL_PRIMARY_NAME_G).Value)
        term = ExtractBeforeSpaceApostrophe(nm)
        If Len(term) > 0 Then dictCatMapPrimaryTerms(term) = True

        altNames = CStr(wsCatMap.Cells(r, CATMAP_COL_ALT_NAMES_J).Value)
        If Len(altNames) > 0 Then
            parts = Split(altNames, ";")
            For Each part In parts
                altTerm = ExtractBeforeSpaceApostrophe(CStr(part))
                If Len(altTerm) > 0 Then dictCatMapAltTerms(altTerm) = True
            Next part
        End If
    Next r

    '========================
    ' 3) Write results to Copy Here column M
    '    Match Copy Here column C against CatMap G first, then fallback to J aliases
    '========================
    lastRow = wsCopy.Cells(wsCopy.Rows.Count, COL_SKU_A).End(xlUp).Row

    For r = 2 To lastRow
        v = CStr(wsCopy.Cells(r, COL_NAME_C).Value)
        If Len(v) = 0 Then
            wsCopy.Cells(r, COL_OUT_M).Value = vbNullString
        Else
            term = ExtractBeforeSpaceApostrophe(v)

            If dictCatMapPrimaryTerms.Exists(term) Then
                wsCopy.Cells(r, COL_OUT_M).Value = "OK"
            ElseIf dictCatMapAltTerms.Exists(term) Then
                wsCopy.Cells(r, COL_OUT_M).Value = "OK"
            Else
                wsCopy.Cells(r, COL_OUT_M).Value = "NO MATCH - CHECK CATEGORIES"
            End If
        End If
    Next r

CleanExit:
    Application.Calculation = xlCalculationAutomatic
    Application.EnableEvents = True
    Application.ScreenUpdating = True
    Exit Sub

CleanFail:
    'Always restore Excel settings
    Resume CleanExit
End Sub

'========================
' Button creator (run once)
'========================
Public Sub Create_CopyHere_RunButton()
    Dim wb As Workbook
    Dim ws As Worksheet
    Dim rng As Range
    Dim shp As Shape
    Dim nm As String: nm = "btnRunParentCat"

    Set wb = ActiveWorkbook
    Set ws = GetSheetLoose(wb, "Copy Here")
    If ws Is Nothing Then
        MsgBox "Can't find sheet 'Copy Here'.", vbExclamation
        Exit Sub
    End If

    Set rng = ws.Range("N1")

    'Delete existing button if present
    On Error Resume Next
    ws.Shapes(nm).Delete
    On Error GoTo 0

    'Create a fixed, horizontal button in row 1
    Set shp = ws.Shapes.AddShape(msoShapeRoundedRectangle, rng.Left + 2, rng.Top + 2, 215, 20)
    shp.Name = nm
    shp.OnAction = "Run_CopyHere_ParentRows_And_CategoryCheck"

    With shp
        .TextFrame2.TextRange.Text = " Run Parent Adder and Cat Match "
        .TextFrame2.Orientation = msoTextOrientationHorizontal
        .TextFrame2.WordWrap = msoFalse
        .TextFrame2.AutoSize = msoAutoSizeNone
        .TextFrame2.MarginLeft = 8
        .TextFrame2.MarginRight = 8
        .TextFrame2.MarginTop = 2
        .TextFrame2.MarginBottom = 2
        .Left = rng.Left + 2
        .Top = ws.Rows(1).Top + 1

        .Fill.ForeColor.RGB = RGB(0, 51, 102) 'dark blue
        .Line.Visible = msoFalse

        With .TextFrame2.TextRange.Font
            .Fill.ForeColor.RGB = RGB(255, 255, 255) 'white
            .Bold = msoTrue
            .Size = 9
        End With

        .Placement = xlMoveAndSize
    End With
End Sub
